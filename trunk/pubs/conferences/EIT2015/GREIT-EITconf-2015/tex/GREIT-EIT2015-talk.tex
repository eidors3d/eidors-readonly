\section{GREIT}
\frame{\frametitle{%
GREIT
}
\begin{equation}
\RB = \argmin_{\footnotesize\RB}\sum_k \norm*{\xT^{(k)} - 
\RB\yB^{(k)}}^2_{\wB^{(k)}}
\end{equation}
\begin{itemize}
	\item $\RB$: Reconstruction matrix
	\item $k$ Training pairs $\{\xT^{(i)},\yB^{(i)}\}$
	\item $\xT^{(i)}$: Desired image of training disturbance $i$
	\item $\yB^{(i)}$: Voltages of training disturbance $i$
\end{itemize}
}
